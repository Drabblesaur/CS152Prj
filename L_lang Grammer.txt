L Language Grammer 

PROGRAM -> FUNCTION

FUNCTION -> FUNCTION IDENTIFIER (DECLARATIONS) { BEGINBODY BEGINLOCAL DECLARATION ENDLOCAL STATEMENT | RETURN ENDBODY }

DECLARATIONS -> DECLARATION DECLARATIONS | EPSILON 
DECLARATION -> IDENTIFIER: INT[NUMBER] | IDENTIFER: INT 

IDENTIFIER -> IDENTIFIER | IDENTIFIER, IDENTIFIER

STATEMENTS -> STATEMENT STATEMENTS | STATEMENT SEMICOLON 
STATEMENT -> VAR | WHILE | DO | IF | ELSE IF | PRINT | PRINTLN 

WHILE -> WHILE       BEGINLOOP STATEMENTS ENDLOOP
DO -> DO  BEGINLOOP STATEMENTS ENDLOOP WHILE 
PRINT -> PRINT VAR
PRINTLN -> PRINTLN VAR 
BOOL -> RELATIONAL | BOOL OR RELATIONAL 

RELATIONAL -> RELATIONALS | RELATIONAL AND RELATIONAL 
REALTIONS -> OP | EXPRESSION 
OP -> EQ | NEQ | GT | GTE | LT | LTE 
EXPRESSION -> MULOP | ADDOP 
ADDOP -> PLUS EXPRESSION | MINUS EXPRESSION | EPSILON 
MULOP -> TERM | TERM MULT MULOP | TERM DIV MULOP | TERM MOD MULOP 

TERM -> MINUS VAR | VAR | MINUS NUMBER | NUMBER | (EXPRESSION) 
VAR -> IDENTIFER | IDENTIFIER[EXPRESSION]
VARS -> VAR | VARS
