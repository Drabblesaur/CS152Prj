L Language Grammer 

PROGRAM -> FUNCTION

FUNCTIONS -> FUNCTION FUNCTIONS | EPSILON 
FUNCTION -> FUNCTION VAR_NAME (DECLARATIONS) {BEGINLOCAL DECLARATION ENDLOCAL STATEMENTS RETURN} 

DECLARATIONS -> DECLARATION DECLARATIONS | EPSILON 
DECLARATION -> TYPE VAR_NAME[NUMBER] | TYPE VAR_NAME 

STATEMENTS -> STATEMENT STATEMENTS | STATEMENT SEMICOLON 
STATEMENT -> SASSIGMENT | SWHILE | SDO | SIF | SELSE-IF | SPRINT | SPRINTLN | SREAD | SRETURN 

SWHILE -> WHILE (RELATIONAL) {STATEMENTS}
SDO -> DO {STATEMENTS} WHILE LPR RELATIONAL RPR
SIF -> IF (RELATIONAL) {STATEMENTS}
SELSE-IF -> ELSE-IF (RELATIONAL){STATEMENTS}
SPRINT -> PRINT {STATEMENTS}
SPRINTLN -> PRINTLN {STATEMENTS}
SREAD -> READ {STATEMENTS}
SRETURN -> RETURN EXPRESSIONS


RELATIONAL -> EXPRESSION COMP EXPRESSION | TRUE | FALSE 
COMP -> EQ | NEQ | GT | GTE | LT | LTE 

EXPRESSION -> MULOP | ADDOP
MULOP -> TERM | TERM MUL MULOP | TERM DIV MULOP | TERM MOD MULOP 
ADDOP -> + EXPRESSION | - EXPRESSION | EPSILON 
TERM -> NUMBER | VAR_NAME
